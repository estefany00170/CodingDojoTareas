#Consulta: recupera todos los usuarios que marcaron como favorito el tercer libro
SELECT nombre, apellido
FROM usuarios u JOIN favoritos f
ON u.id = f.id_usuario
LEFT JOIN libros l
ON l.id = f.id_libro
WHERE id_libro = 3;
